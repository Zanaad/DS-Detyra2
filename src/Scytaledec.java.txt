public static String decrypt(String cipherText, int key) {
    int col = key;
    int row = (int) Math.ceil((double) cipherText.length() / key);
    char[][] cipherMatrix = new char[row][col];

    // Mbushja e matricës kolonë për kolonë
    int index = 0;
    for (int r = 0; r < row; r++) {
        for (int c = 0; c < col; c++) {
            cipherMatrix[r][c] = cipherText.charAt(index);
            index++;
        }
    }

    // Leximi i matricës rresht për rresht për të marrë plaintext
    StringBuilder plaintextBuilder = new StringBuilder();
    for (int c = 0; c < col; c++) {
        for (int r = 0; r < row; r++) {
            plaintextBuilder.append(cipherMatrix[r][c]);
        }
    }
    return plaintextBuilder.toString().trim(); // Heq hapësirat e panevojshme në fund
}
